# Flutter Navigation

GoRouter for declarative, URL-based navigation with type safety.

## Setup (CRITICAL)

```yaml
dependencies:
  go_router: ^13.0.0
```

```dart
// app/router.dart
@riverpod
GoRouter router(RouterRef ref) {
  final authState = ref.watch(authStateProvider);

  return GoRouter(
    initialLocation: '/home',
    debugLogDiagnostics: kDebugMode,
    redirect: (context, state) {
      final isLoggedIn = authState.valueOrNull != null;
      final isAuthRoute = state.matchedLocation.startsWith('/auth');

      if (!isLoggedIn && !isAuthRoute) return '/auth/login';
      if (isLoggedIn && isAuthRoute) return '/home';
      return null;
    },
    routes: $appRoutes,  // Generated by go_router_builder
  );
}
```

## Route Definitions (CRITICAL)

Using `go_router_builder` for type safety:

```dart
// app/routes.dart
part 'routes.g.dart';

@TypedGoRoute<HomeRoute>(path: '/home')
@immutable
class HomeRoute extends GoRouteData {
  const HomeRoute();

  @override
  Widget build(BuildContext context, GoRouterState state) => const HomeScreen();
}

@TypedGoRoute<PostDetailRoute>(path: '/posts/:postId')
@immutable
class PostDetailRoute extends GoRouteData {
  const PostDetailRoute({required this.postId});
  final String postId;

  @override
  Widget build(BuildContext context, GoRouterState state) {
    return PostDetailScreen(postId: postId);
  }
}

@TypedGoRoute<LoginRoute>(path: '/auth/login')
@immutable
class LoginRoute extends GoRouteData {
  const LoginRoute({this.redirect});
  final String? redirect;  // Query param

  @override
  Widget build(BuildContext context, GoRouterState state) => const LoginScreen();
}
```

## Navigation (HIGH)

```dart
// Type-safe navigation — compile error if params are wrong
context.go(const HomeRoute());
context.push(PostDetailRoute(postId: post.id));
context.pop();

// With query params
context.push(LoginRoute(redirect: state.uri.toString()));

// Replace current route (no back button)
context.replace(const HomeRoute());

// Named navigation (without go_router_builder)
context.goNamed('post-detail', pathParameters: {'postId': post.id});
```

## Nested Navigation (HIGH)

```dart
// Shell route for bottom navigation
@TypedShellRoute<AppShellRoute>(routes: [
  TypedGoRoute<HomeRoute>(path: '/home'),
  TypedGoRoute<SearchRoute>(path: '/search'),
  TypedGoRoute<ProfileRoute>(path: '/profile'),
])
@immutable
class AppShellRoute extends ShellRouteData {
  const AppShellRoute();

  @override
  Widget builder(BuildContext context, GoRouterState state, Widget navigator) {
    return AppShell(child: navigator);
  }
}

// AppShell with bottom nav
class AppShell extends StatelessWidget {
  const AppShell({super.key, required this.child});
  final Widget child;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: child,
      bottomNavigationBar: BottomNavigationBar(
        currentIndex: _currentIndex(context),
        onTap: (index) => _onTap(context, index),
        items: const [
          BottomNavigationBarItem(icon: Icon(Icons.home), label: 'Home'),
          BottomNavigationBarItem(icon: Icon(Icons.search), label: 'Search'),
          BottomNavigationBarItem(icon: Icon(Icons.person), label: 'Profile'),
        ],
      ),
    );
  }

  int _currentIndex(BuildContext context) {
    final location = GoRouterState.of(context).matchedLocation;
    if (location.startsWith('/search')) return 1;
    if (location.startsWith('/profile')) return 2;
    return 0;
  }

  void _onTap(BuildContext context, int index) {
    switch (index) {
      case 0: const HomeRoute().go(context);
      case 1: const SearchRoute().go(context);
      case 2: const ProfileRoute().go(context);
    }
  }
}
```

## Route Guards (MEDIUM)

```dart
// Per-route redirect
@TypedGoRoute<AdminRoute>(path: '/admin')
@immutable
class AdminRoute extends GoRouteData {
  const AdminRoute();

  @override
  String? redirect(BuildContext context, GoRouterState state) {
    final user = context.read(currentUserProvider);
    if (user == null) return const LoginRoute().location;
    if (!user.isAdmin) return const HomeRoute().location;
    return null;
  }

  @override
  Widget build(BuildContext context, GoRouterState state) => const AdminScreen();
}
```

## Rules

- Always use `go_router_builder` for type-safe navigation — never string paths
- Always implement global `redirect` for auth guards — never check auth in individual screens
- Use `context.go()` for tab switches, `context.push()` for drill-down navigation
- Never use `Navigator.push()` — always use GoRouter for consistent deep link support
